Idk where else to put it, but here's the SQL code that I (Harry) have used for setting up the database

NOTE: I'm doing all this in the default "PostgreSQL 18" server that came with my pgadmin install.

----DATABASE----

CREATE DATABASE "DSSwebsiteDB"
    WITH
    OWNER = postgres     //this is the default username that's attached to the server and idk if i can change it or if its even needed
    TEMPLATE = postgres
    ENCODING = 'UTF8'
    LOCALE_PROVIDER = 'libc'
    CONNECTION LIMIT = -1
    IS_TEMPLATE = False;

----SCHEMA----

CREATE SCHEMA gameBlog
    AUTHORIZATION postgres;

NOTE: you might have to go through pgAdmin's creating-menu for database and schema
For database: set the name, then in Definition, set the template to "postgres" (if you get an arror about daticulocale not existing, ignore it, the database created anyway)
for schema, literally just set the name

---The part below is guaranteed to be copy-paste-eable---

SET search_path TO "gameBlog";

----USER TABLE + ENUM----

CREATE TYPE "userTypeEnum" AS ENUM
    ('Standard', 'Admin');

ALTER TYPE "userTypeEnum"
    OWNER TO postgres;

----

CREATE TABLE "UsersTable"
(
    "userID" integer NOT NULL,
    "userName" character varying(32) NOT NULL,
    "userEmail" character varying(320) NOT NULL,
    "userPassHash" character varying(64) NOT NULL,
    "userType" "userTypeEnum" NOT NULL,
    PRIMARY KEY ("userID")
);

ALTER TABLE IF EXISTS "UsersTable"
    OWNER to postgres;
    
ALTER TABLE "UsersTable"
ADD CONSTRAINT "userNameUnique" UNIQUE ("userName");
ALTER TABLE "UsersTable"
ADD CONSTRAINT "userEmailUnique" UNIQUE ("userEmail");

----PASSWORD SALTS TABLE---- //stored in a different table for a bit of extra security

CREATE TABLE "UserPassSaltsTable"
(
    "userID" integer NOT NULL references "UsersTable"("userID"),
    "passSalt" character varying(16) NOT NULL,
    PRIMARY KEY ("userID")
);

ALTER TABLE IF EXISTS "UserPassSaltsTable"
    OWNER to postgres;

----GAME TABLE----

CREATE TABLE "GamesTable"
(
    "gameID" integer NOT NULL,
    "gameName" character varying(128) NOT NULL,
    "gameDesc" character varying(1024),
    PRIMARY KEY ("gameID")
);

ALTER TABLE IF EXISTS "GamesTable"
    OWNER to postgres;

----POST TABLE + ENUM----

CREATE TYPE "postTypeEnum" AS ENUM
    ('Standard', 'Review', 'BugReport', 'DevLog');

ALTER TYPE "postTypeEnum"
    OWNER TO postgres;

----

CREATE TABLE "PostsTable"
(
    "postID" integer NOT NULL,
    "userID" integer NOT NULL references "UsersTable"("userID"),
    "gameID" integer references "GamesTable"("gameID"),
    "postType" "postTypeEnum" NOT NULL,
    "postTitle" character varying(256) NOT NULL,
    "postContent" character varying(2048),
    "postReviewScore" integer,
    PRIMARY KEY ("postID")
);

ALTER TABLE IF EXISTS "PostsTable"
    OWNER to postgres;
